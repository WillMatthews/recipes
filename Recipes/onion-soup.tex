\food{French Onion Soup}
\preptime{30 minutes}
\cooktime{3 hours}
\people{4-6}

\recipe{
    \unit[1]{kg} & Onions\\
    \unit[4]{tbsp} & Butter\\
    \unit[6]{tbsp} & Olive Oil\\
    \unit[1]{tsp} & Salt\\
    \unit[1]{tsp} & Sugar\\
    \unit[2]{cloves} & Garlic\\
    \unit[2]{} & Bay Leaves\\
    \unit[1]{tsp} & Thyme\\
    \unit[1]{tsp} & Pepper\\
    \unit[1]{cup} & White Wine\\
    \unit[4]{Cups} & Chicken Stock\\
    \unit[2]{Cups} & Gruyere Cheese\\
    \unit[1]{Loaf} & Crusty French Bread\\
}{
    \item Slice the onions thinly.
    \item In a large pot, melt the butter and olive oil together.
    \item Add the onions, salt, and sugar to the pot and cook on low heat for 1.5 hours, stirring occasionally.
    \item The onions should be a deep golden brown when done.
    \item Add the garlic, bay leaves, thyme, and pepper to the pot and cook for another 30 minutes.
    \item Add the white wine to the pot and cook for 5 minutes.
    \item Add the chicken stock to the pot and cook for 30 minutes.
    \item Preheat the oven to 200\degree C.
    \item Slice the bread into 1 cm slices and toast in the oven until golden brown.
    \item Grate cheese onto the bread slices, and return to the oven until the cheese is melted and bubbly.
    \item Ladle the soup into oven-safe bowls, top with a slice of bread, and sprinkle with more cheese (because life is short).
}
