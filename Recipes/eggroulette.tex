\food{Egg Roulette}
\preptime{1 min} \people{n>2}

\recipe{
        n-1 & Large eggs (hard boiled)\\
        1 & Century Egg\\
}{
    \item Place n-1 large hard boiled eggs, and one century egg in an opaque bag.
    \item Each person draws an egg from the bag.
    \item Everyone eats their egg.
}

\info{Century eggs are a Chinese delicacy, typically used to flavour congee. This game works best if you and all people at the table hate century eggs. Conversely, if you enjoy century eggs like Jenny, you're always a winner.}
