\food{Broccoli Cheese Bake}
\preptime{10 minutes}
\cooktime{20 minutes}
\baketime{180\degree C}
\people{4}

\recipe{
    \unit{1}{kg} & Broccoli\\
    \unit{50}{g} & Butter\\
    \unit{50}{g} & Flour\\
    \unit{500}{ml} & Milk\\
    \unit{200}{g} & Cheddar Cheese\\
}{
    \item Begin by boiling the broccoli until it is soft.
    \item While the broccoli is boiling, melt the butter in a saucepan.
    \item Add the flour to the butter and whisk until it forms a roux.
    \item Slowly add the milk to the roux, whisking constantly.
    \item Add the cheese to the sauce and stir until it is melted.
    \item Drain the broccoli and add it to the cheese sauce.
    \item Pour the mixture into a baking dish and bake at 180\degree C for 20 minutes.
}
