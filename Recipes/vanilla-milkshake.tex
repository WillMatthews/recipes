\food{Vanilla Milkshake}
\preptime{5 minutes}
\people{2}

\recipe{
    \unit[500]{ml} & Milk (The more fat, the better)\\
    \unit[400]{g} & Vanilla Ice Cream\\
    \unit[3]{tsp} & Vanilla Extract\\
     & Whipped Cream
}{
    \item Blend the milk, ice cream, and vanilla extract until smooth.
    \item Pour into a glass and top with whipped cream.
    \item Ta-daaah! It's that simple. Enjoy the coronary.
}
